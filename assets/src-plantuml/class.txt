  finalProductScore: Review

  tags: List<String>

  email: String
  password: Int
  isModerator: Boolean

  changeUsername()
  changePassword()
  changeEmail()
}

  history: List<Post>

  changeName()
  changeContact()
  changeAboutMe()
}

class Review {
  grade: Float
  subject: String
  body: String

  editSubject()
  editBody()
